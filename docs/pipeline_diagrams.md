# Диаграммы конвейера Audit Manager

> Актуально на 2026-03-25. Модели: OpenRouter (GPT-5.4, Gemini 3.1 Pro) + Claude Opus.

## 1. Основной конвейер аудита

```mermaid
flowchart TD
    subgraph S01["01 — Вырезка блоков из PDF"]
        PDF(["Исходный PDF"]) --> CROP["Скрипт вырезки блоков\n─────────\nСкачивание фрагментов\nчертежей по координатам\nиз результатов OCR\nМасштаб до 1500 пикселей"]
        CROP --> BLOCKS(["Изображения блоков\n+ индекс блоков"])
    end

    subgraph S02["02 — Граф документа"]
        MD(["Текст документа\nиз OCR-сервиса"]) --> DG_BUILD["Построение графа\n─────────\nРазбор текста в граф страниц\nОбогащение данными\nиз индекса блоков"]
        BLOCKS --> DG_BUILD
        DG_BUILD --> DG(["Граф документа"])
    end

    subgraph S03["03 — Анализ текста"]
        DG --> TA["ИИ: анализ текста\n(GPT-5.4)\n─────────\nПроверка таблиц\nПерекрёстная сверка\nНормативные ссылки\nПриоритизация блоков"]
        TA --> T1(["Результат\nанализа текста"])
    end

    subgraph S04["04 — Анализ чертежей"]
        T1 --> BATCH["Группировка блоков\n─────────\nРаспределение\nпо приоритетам"]
        BLOCKS --> BATCH
        DG --> BATCH
        BATCH --> CLAUDE_N["ИИ: анализ блоков\n(Gemini 3.1 Pro)\n─────────\nДо 3 сессий параллельно\nКонтекст страницы\nдля каждого блока"]
        CLAUDE_N --> MERGE["Слияние результатов\n─────────\nОбъединение результатов\nвсех сессий анализа"]
        MERGE --> T2(["Результат\nанализа чертежей"])
    end

    subgraph S05["05 — Свод замечаний"]
        T1 --> FINDINGS["ИИ: объединение\n(Claude Opus)\n─────────\nОбъединение текстовых\nи графических замечаний\nМежстраничная сверка\nУдаление дубликатов"]
        T2 --> FINDINGS
        FINDINGS --> T3(["Итоговые замечания"])
    end

    subgraph S06["06 — Параллельные потоки"]
        T3 --> PAR_C(["Поток 1:\nПроверка замечаний"])
        T3 --> PAR_N(["Поток 2:\nПроверка норм"])
        T3 --> PAR_O(["Поток 3:\nОптимизация"])
    end

    subgraph S06a["06а — Проверка замечаний"]
        PAR_C --> CRITIC["ИИ: критик\n(GPT-5.4)\n─────────\n5 проверок обоснованности\nПорции по 50 замечаний\nДо 3 порций параллельно"]
        CRITIC --> REVIEW(["Вердикты проверки"])
        REVIEW -->|"есть проблемы"| CORRECTOR["ИИ: корректор\n(GPT-5.4)\n─────────\nИсправление или удаление\nнеобоснованных замечаний"]
        CORRECTOR --> T3_FIX(["Замечания\nисправленные"])
        REVIEW -->|"всё в порядке"| SKIP(["Пропуск"])
    end

    subgraph S06b["06б — Проверка нормативов"]
        PAR_N --> NORMS["Проверка норм\n─────────\n1. Поиск ссылок на нормы\n2. Проверка по кешу\n3. ИИ-поиск устаревших\n4. Объединение результатов"]
        NORMS --> NC(["Результат\nпроверки норм"])
        NC -->|"есть устаревшие"| NFIX["ИИ: обновление ссылок\n(GPT-5.4)\n─────────\nПересмотр замечаний\nс устаревшими нормами"]
        NFIX --> T3_NFIX(["Замечания\nобновлённые"])
        NC -->|"все актуальны"| NSKIP(["Пропуск"])
    end

    subgraph S06c["06в — Оптимизация"]
        PAR_O --> OPT["ИИ: оптимизация\n(Gemini 3.1 Pro)\n─────────\nЗамена на аналоги\nУпрощение монтажа\nУчёт списка поставщиков\nТаймаут 60 мин"]
        OPT --> O1(["Предложения\nпо оптимизации"])
        O1 --> OCRIT["ИИ: критик оптимизации\n(GPT-5.4)\n─────────\nПроверка поставщиков\nРеалистичность экономии\nКонфликты с замечаниями"]
        OCRIT --> OREV(["Вердикты проверки\nоптимизации"])
        OREV -->|"есть проблемы"| OCORR["ИИ: корректор\n(GPT-5.4)\n─────────\nИсправление или удаление\nнеобоснованных предложений"]
        OCORR --> O1_FIX(["Оптимизация\nисправленная"])
        OREV -->|"всё в порядке"| OSKIP(["Пропуск"])
    end

    subgraph S07["07 — Итоговый отчёт"]
        T3_FIX --> EXCEL["Формирование отчёта\n─────────\nЗамечания + Нормы +\nОптимизация\nв итоговый Excel-файл"]
        NC --> EXCEL
        O1_FIX --> EXCEL
        EXCEL --> XLSX(["Отчёт Excel"])
    end
```

## 2. Проверка нормативных документов

```mermaid
flowchart TD
    subgraph S01["01 — Извлечение ссылок на нормы"]
        START(["Итоговые замечания"]) --> REGEX["Поиск ссылок\n─────────\nИщет упоминания:\nСП, ГОСТ, ПУЭ,\nСНиП, ФЗ"]
        REGEX --> LIST(["Список найденных норм"])
    end

    subgraph S02["02 — Проверка по базе"]
        LIST --> DB{"База нормативов\n439 документов"}
        DB -->|"Есть в базе, свежий"| DET["Быстрая проверка\n─────────\nСтатус из кеша\nБез вызова ИИ\nСрок хранения 180 дней"]
        DB -->|"Устарел (>180 дней)"| WS["ИИ-поиск в интернете\n(Claude Opus)\n─────────\nПоиск актуального статуса\nПорции по 15 норм\nДо 3 порций параллельно"]
        DB -->|"Нет в базе"| WS
    end

    subgraph S03["03 — Объединение результатов"]
        DET --> NC(["Результат проверки\n─────────\nОбъединение данных\nиз кеша и ИИ-поиска"])
        WS --> LLM(["Результат ИИ-поиска"])
        LLM -->|"Объединение"| NC
    end

    subgraph S04["04 — Проверка точности цитат"]
        NC --> PARA{"Уверенность < 80%?"}
        PARA -->|"Да"| PCHECK["ИИ-поиск цитаты\n─────────\nПоиск конкретного\nпункта нормы\nв интернете"]
        PARA -->|"Нет"| OK(["Норма подтверждена"])
        PCHECK --> PCACHE(["Кеш проверенных\nцитат норм"])
        PCHECK --> VERIFIED{"Цитата верна?"}
        VERIFIED -->|"Да"| OK
        VERIFIED -->|"Нет"| REVISION(["Отправить\nна пересмотр"])
    end

    subgraph S05["05 — Исправление устаревших ссылок"]
        NC --> STALE{"Есть устаревшие?"}
        STALE -->|"Да"| NFIX["ИИ: обновление ссылок\n(GPT-5.4)\n─────────\nПересмотр замечаний\nЗамена на актуальные нормы"]
        NFIX --> FIXED(["Замечания обновлены"])
        STALE -->|"Нет"| DONE(["Пересмотр не нужен"])
    end
```

## 3. Анализ чертежей (детали)

```mermaid
flowchart LR
    subgraph S_CROP["Вырезка"]
        OCR(["Результат OCR\n─────────\nКоординаты\nфрагментов чертежей"]) --> CROP["Скрипт вырезки\n─────────\nМасштаб до 1500px\nФильтр мелких блоков"]
        CROP --> PNG(["Изображения\nблоков"])
        CROP --> IDX(["Индекс блоков\n─────────\nНомер блока, страница\nОписание, размер"])
    end

    subgraph S_BATCH["Группировка"]
        PNG --> BAT["Формирование пакетов\n─────────\nГруппировка блоков\nпо приоритетам из\nанализа текста"]
        IDX --> BAT
        BAT --> BB(["Пакеты блоков"])
    end

    subgraph S_ANALYSIS["Анализ ИИ"]
        BB --> S1["Gemini 3.1 Pro\nсессия 1"]
        BB --> S2["Gemini 3.1 Pro\nсессия 2"]
        BB --> S3["Gemini 3.1 Pro\nсессия 3"]
        SN["─────────\nТаймаут 10 мин на пакет\nОшибка: разбить пополам\nПревышение лимита: повтор"]
    end

    subgraph S_MERGE["Слияние"]
        S1 --> MRG["Объединение результатов\n─────────\nСлияние и очистка\nвсех результатов сессий"]
        S2 --> MRG
        S3 --> MRG
        MRG --> RESULT(["Результат\nанализа чертежей"])
    end
```

## 4. Проверка замечаний (Критик → Корректор)

```mermaid
flowchart TD
    subgraph S01["01 — Подготовка"]
        F(["Итоговые замечания"]) --> PRE["Предварительная привязка\n─────────\nОбогащение замечаний\nкандидатами на блоки\nдо запуска критика"]
    end

    subgraph S02["02 — Разбиение на порции"]
        PRE --> CHUNK{"Замечаний > 50?"}
        CHUNK -->|"Да"| SPLIT["Разбить на порции\nпо 50 замечаний"]
        CHUNK -->|"Нет"| SINGLE["Одна порция"]
    end

    subgraph S03["03 — Критик (5 проверок)"]
        SPLIT --> C["ИИ: критик (GPT-5.4)\n─────────\nДо 3 порций параллельно\nТаймаут 10 мин на порцию"]
        SINGLE --> C
        C --> C1["1. Есть доказательства?\n─────────\nСсылки на блоки\nили фрагменты текста"]
        C --> C2["2. Блоки существуют?\n─────────\nПроверка по результатам\nанализа чертежей"]
        C --> C3["3. Доказательства\nсоответствуют смыслу?\n─────────\nСемантическая проверка"]
        C --> C4["4. Страница верна?\n─────────\nСверка с графом\nдокумента"]
        C --> C5["5. Нет противоречий?\n─────────\nПроверка по тексту\nисходного документа"]
    end

    subgraph S04["04 — Вердикты и исправление"]
        C1 --> V{"Итог проверки"}
        C2 --> V
        C3 --> V
        C4 --> V
        C5 --> V
        V -->|"всё верно"| PASS(["Замечание\nподтверждено"])
        V -->|"нет доказательств"| FIX["ИИ: корректор\n(GPT-5.4)\n─────────\nЗапускается только\nпри найденных\nпроблемах"]
        V -->|"блок не найден"| FIX
        V -->|"слабые доказательства"| FIX
        V -->|"неверная страница"| FIX
        V -->|"противоречит тексту"| FIX
    end

    subgraph S05["05 — Результат"]
        FIX --> FIXED(["Замечания\nисправленные"])
        FIX --> SIGNAL(["Сигнал готовности\n─────────\nУведомление критика\nоптимизации о готовности\nзамечаний"])
    end
```
